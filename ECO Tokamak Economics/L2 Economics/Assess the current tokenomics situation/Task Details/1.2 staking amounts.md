- example- staking

```shell
stakeds(
      where: {
        transaction_: {
          blockNumber_gte: $fromBlock
          blockNumber_lte: $toBlock
        }
      }
      orderBy: timestamp
      orderDirection: desc
      first: $first
    ) {
      id
      amount
      timestamp
      user {
        id
      }
      candidate {
        name
      }
      transaction {
        id
        blockNumber
      }
    } 
```

- example- withdrawal 

```shell
{
  unstakeds(
    where: {
      transaction_: {
        blockNumber_gte: 23775621
        blockNumber_lte: 23775630
      }
    }
    orderBy: timestamp
    orderDirection: desc
    first: 10
  ) {
    id
    amount
    timestamp
    eventName
   	transaction{
      blockNumber
      id
    }
  }
}
  
```
