## Query to extract `Mint` / `Burn` / `Swap` events from the WETH-WTON pool on Uniswap V3

```sql
  WITH pool_flows AS (
      SELECT evt_block_time, amount0, amount1 FROM uniswap_v3_ethereum.UniswapV3Pool_evt_Mint
      WHERE contract_address = 0xC29271E3a68A7647Fd1399298Ef18FeCA3879F59
      UNION ALL
      SELECT evt_block_time, -amount0, -amount1 FROM uniswap_v3_ethereum.UniswapV3Pool_evt_Burn
      WHERE contract_address = 0xC29271E3a68A7647Fd1399298Ef18FeCA3879F59
      UNION ALL
      SELECT evt_block_time, amount0, amount1 FROM uniswap_v3_ethereum.UniswapV3Pool_evt_Swap
      WHERE contract_address = 0xC29271E3a68A7647Fd1399298Ef18FeCA3879F59
  )

  SELECT
      DATE_TRUNC('day', evt_block_time) AS date,
      SUM(CAST(amount0 AS DOUBLE)) / 1e18 AS net_flow_weth,
      SUM(CAST(amount1 AS DOUBLE)) / 1e27 AS net_flow_wton
  FROM pool_flows
  GROUP BY 1
  ORDER BY 1 DESC
```
