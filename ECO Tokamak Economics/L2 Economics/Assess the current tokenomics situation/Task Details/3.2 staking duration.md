[[UPDATED: average staking duration using Proportional/Weighted Average matching logic]]

Demo of the calculation for staking duration using available [extracted staking log](https://github.com/tokamak-network/tools/tree/master/get_all_transactions/logs_events)

Code, original data file & details can also be found [here](https://drive.google.com/drive/folders/1sSTT7qahFA8d4NyE3D9UM0xVgeArX-yv?usp=drive_link) 

- Simple matching logic: Match each stake with the next available unstake by the same depositor. For every 'Deposited' event, look for the first 'Unstaking' event that occurred after it and calculate the duration between the two events. Then remove that 'Unstaking' event from the pool of available unstakes.
- This is a heuristic and assumes a simple FIFO (First-In, First-Out) staking model: it assumes that the oldest remaining stake is the one that is unstaked next.

The data was collected over a period spanning from **September 11, 2020, to September 28, 2023**.

Calculating avg staking time

```python
import pandas as pd
from datetime import timedelta
import csv

def calculate_staking_data(file_path):
    # Load the CSV file
    df = pd.read_csv(file_path)

    # Convert Timestamp to datetime objects
    df['Timestamp'] = pd.to_datetime(df['Timestamp'])

    # Filter for relevant event types and sort by timestamp
    df_events = df[df['EventType'].isin(['Deposited', 'Unstaking'])].sort_values(by='Timestamp')

    # Group by the staker's address (Depositor)
    grouped = df_events.groupby('Depositor')

    staking_durations = []

    # Iterate over each staker
    for depositor, events in grouped:
        # Separate stake (Deposited) and unstake (Unstaking) events
        stakes = events[events['EventType'] == 'Deposited'].sort_values(by='Timestamp')
        unstakes = events[events['EventType'] == 'Unstaking'].sort_values(by='Timestamp')

        # Use a list of available unstake timestamps for matching
        available_unstakes = list(unstakes['Timestamp'])
        
        for stake_time in stakes['Timestamp']:
            # Find the first unstake time that is after the current stake time
            match_index = -1
            for i, unstake_time in enumerate(available_unstakes):
                if unstake_time > stake_time:
                    match_index = i
                    break
            
            if match_index != -1:
                # A match is found, calculate duration and remove the unstake event from the available pool
                unstake_time = available_unstakes.pop(match_index)
                duration = unstake_time - stake_time
                staking_durations.append(duration)

    if not staking_durations:
        return "No complete staking cycles (Deposited followed by Unstaking) found in the data."

    # Convert durations to total seconds
    staking_durations_seconds = [d.total_seconds() for d in staking_durations]

    # Save the durations to a new CSV file
    output_file = 'staking_durations_seconds.csv'
    with open(output_file, 'w', newline='') as f:
        writer = csv.writer(f)
        writer.writerow(['duration_seconds'])
        writer.writerows([[s] for s in staking_durations_seconds])

    # Calculate the average duration
    total_duration = sum(staking_durations, timedelta(0))
    average_duration = total_duration / len(staking_durations)

    # Convert the average duration to a human-readable format (e.g., days, hours, minutes)
    total_seconds = average_duration.total_seconds()
    days = int(total_seconds // (24 * 3600))
    hours = int((total_seconds % (24 * 3600)) // 3600)
    minutes = int((total_seconds % 3600) // 60)
    seconds = int(total_seconds % 60)

    return {
        "average_duration_timedelta": average_duration,
        "average_duration_human_readable": f"{days} days, {hours} hours, {minutes} minutes, {seconds} seconds",
        "total_cycles": len(staking_durations),
        "output_file": output_file
    }

if __name__ == "__main__":
    file_path = '/home/ubuntu/upload/10837675_18231453.csv'
    result = calculate_staking_data(file_path)
    
    if isinstance(result, str):
        print(result)
    else:
        print(f"Total staking cycles analyzed: {result['total_cycles']}")
        print(f"Average staking time (timedelta): {result['average_duration_timedelta']}")
        print(f"Average staking time: {result['average_duration_human_readable']}")
        print(f"Staking durations saved to: {result['output_file']}")
```

```python
import pandas as pd
import matplotlib.pyplot as plt
import numpy as np

def create_histogram(data_file, output_file):
    # Load the staking durations in seconds
    df = pd.read_csv(data_file)
    
    # Convert duration from seconds to days for better visualization
    df['duration_days'] = df['duration_seconds'] / (24 * 3600)
    
    # Calculate the average duration in days for the vertical line
    average_duration_days = df['duration_days'].mean()

    # Set up the plot
    plt.figure(figsize=(12, 6))
    
    # Create the histogram
    # Use a number of bins that is appropriate for the data range.
    # We can use the square root of the number of data points as a starting point for the number of bins.
    n_bins = int(np.sqrt(len(df)))
    
    plt.hist(df['duration_days'], bins=n_bins, color='skyblue', edgecolor='black', alpha=0.7)
    
    # Add a vertical line for the average staking time
    plt.axvline(average_duration_days, color='red', linestyle='dashed', linewidth=1.5, 
                label=f'Average: {average_duration_days:.2f} days')

    # Add labels and title
    plt.title('Distribution of Staking Times', fontsize=16)
    plt.xlabel('Staking Duration (Days)', fontsize=12)
    plt.ylabel('Frequency (Number of Staking Cycles)', fontsize=12)
    plt.legend()
    plt.grid(axis='y', alpha=0.5)
    
    # Save the plot
    plt.savefig(output_file)
    plt.close()
    
    print(f"Histogram saved to {output_file}")

if __name__ == "__main__":
    data_file = 'staking_durations_seconds.csv'
    output_file = 'staking_time_histogram.png'
    create_histogram(data_file, output_file)
```

```python
import pandas as pd
import matplotlib.pyplot as plt
import numpy as np

def create_boxplot(data_file, output_file):
    # Load the staking durations in seconds
    df = pd.read_csv(data_file)
    
    # Convert duration from seconds to days for better visualization
    df['duration_days'] = df['duration_seconds'] / (24 * 3600)
    
    # Set up the plot
    plt.figure(figsize=(10, 6))
    
    # Create the box plot
    # We use vert=False to make it a horizontal box plot for better readability
    plt.boxplot(df['duration_days'], vert=False, patch_artist=True, 
                boxprops=dict(facecolor='lightblue'),
                medianprops=dict(color='red', linewidth=2))
    
    # Add labels and title
    plt.title('Box Plot of Staking Times', fontsize=16)
    plt.xlabel('Staking Duration (Days)', fontsize=12)
    plt.yticks([1], ['Staking Time']) # Set a single y-tick for the horizontal plot
    
    # Add text for key statistics
    median = df['duration_days'].median()
    q1 = df['duration_days'].quantile(0.25)
    q3 = df['duration_days'].quantile(0.75)
    
    plt.text(median, 1.1, f'Median: {median:.2f} days', ha='center', color='red', weight='bold')
    plt.text(q1, 1.05, f'Q1: {q1:.2f} days', ha='center')
    plt.text(q3, 1.05, f'Q3: {q3:.2f} days', ha='center')

    # Adjust layout to prevent labels from being cut off
    plt.tight_layout()
    
    # Save the plot
    plt.savefig(output_file)
    plt.close()
    
    print(f"Box plot saved to {output_file}")

if __name__ == "__main__":
    data_file = 'staking_durations_seconds.csv'
    output_file = 'staking_time_boxplot.png'
    create_boxplot(data_file, output_file)
```
