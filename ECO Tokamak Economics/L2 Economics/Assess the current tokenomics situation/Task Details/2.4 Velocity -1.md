Code to extract volume data from `Upbit`, `Bithumb`, and `DigiFinex` exchanges—confirmed to provide daily volume information—and generate it as a CSV file.

code
```python
import requests
import csv
from datetime import datetime, timedelta
import time

def get_upbit_candles():
    """
    Upbit 일봉 캔들 데이터 조회
    - 최대 200개 캔들/요청
    - 페이지네이션으로 더 많은 데이터 조회 가능
    """
    candles = []
    url = "https://api.upbit.com/v1/candles/days"
    to = None  # 가장 최근부터 시작

    while True:
        params = {"market": "KRW-TOKAMAK", "count": 200}
        if to:
            params["to"] = to

        response = requests.get(url, params=params)
        data = response.json()

        if not data or "error" in str(data):
            break

        for candle in data:
            candles.append({
                "date": candle["candle_date_time_kst"][:10],
                "open": candle["opening_price"],
                "high": candle["high_price"],
                "low": candle["low_price"],
                "close": candle["trade_price"],
                "volume": candle["candle_acc_trade_volume"],
                "trade_value": candle["candle_acc_trade_price"],
            })

        if len(data) < 200:
            break

        # 다음 페이지를 위해 마지막 캔들의 시간 사용
        to = data[-1]["candle_date_time_utc"]
        time.sleep(0.1)  # API 제한 방지

    return {"exchange": "Upbit", "symbol": "TOKAMAK", "candles": candles}


def get_bithumb_candles():
    """
    Bithumb 일봉 캔들 데이터 조회
    - 24h 간격으로 조회
    - 약 1500개 정도 반환
    """
    url = "https://api.bithumb.com/public/candlestick/TOKAMAK_KRW/24h"
    response = requests.get(url)
    data = response.json()

    candles = []
    if data.get("status") == "0000":
        raw_candles = data.get("data", [])
        for candle in raw_candles:
            # [시간, 시가, 종가, 고가, 저가, 거래량]
            timestamp = int(candle[0]) / 1000
            date = datetime.fromtimestamp(timestamp).strftime("%Y-%m-%d")
            candles.append({
                "date": date,
                "open": candle[1],
                "high": candle[3],
                "low": candle[4],
                "close": candle[2],
                "volume": candle[5],
                "trade_value": None,
            })

    return {"exchange": "Bithumb", "symbol": "TOKAMAK", "candles": candles}


def get_biconomy_candles():
    """
    Biconomy 일봉 캔들 데이터 조회
    - period: 1day
    - 최대 1000개 캔들
    """
    url = "https://www.biconomy.com/api/v1/kline"
    params = {
        "symbol": "TOKAMAK_USDT",
        "period": "1day",
        "size": 1000
    }
    response = requests.get(url, params=params)
    data = response.json()

    candles = []
    if data.get("code") == 0:
        raw_candles = data.get("data", [])
        for candle in raw_candles:
            timestamp = int(candle.get("id", 0))
            date = datetime.fromtimestamp(timestamp).strftime("%Y-%m-%d")
            candles.append({
                "date": date,
                "open": candle.get("open"),
                "high": candle.get("high"),
                "low": candle.get("low"),
                "close": candle.get("close"),
                "volume": candle.get("vol"),
                "trade_value": None,
            })

    return {"exchange": "Biconomy", "symbol": "TOKAMAK", "candles": candles}


def get_digifinex_candles():
    """
    DigiFinex 일봉 캔들 데이터 조회
    - period: 1D (일봉)
    - 최대 500개 캔들/요청
    """
    candles = []
    url = "https://openapi.digifinex.com/v3/kline"

    # 최대한 과거 데이터부터 가져오기
    end_time = int(datetime.now().timestamp())
    start_time = int((datetime.now() - timedelta(days=500)).timestamp())

    params = {
        "symbol": "tokamak_usdt",
        "period": "1D",
        "start_time": start_time,
        "end_time": end_time,
        "limit": 500
    }
    response = requests.get(url, params=params)
    data = response.json()

    if data.get("code") == 0:
        raw_candles = data.get("data", [])
        for candle in raw_candles:
            # [timestamp, volume, close, high, low, open]
            timestamp = int(candle[0])
            date = datetime.fromtimestamp(timestamp).strftime("%Y-%m-%d")
            candles.append({
                "date": date,
                "open": candle[5],
                "high": candle[3],
                "low": candle[4],
                "close": candle[2],
                "volume": candle[1],
                "trade_value": None,
            })

    return {"exchange": "DigiFinex", "symbol": "TOKAMAK", "candles": candles}


def save_to_csv(result, filename):
    """캔들 데이터를 CSV 파일로 저장"""
    candles = result.get("candles", [])
    if not candles:
        print(f"  {result['exchange']}: 데이터 없음")
        return

    # 날짜 오름차순 정렬
    candles.sort(key=lambda x: x["date"])

    with open(filename, "w", newline="", encoding="utf-8") as f:
        writer = csv.writer(f)
        writer.writerow(["date", "open", "high", "low", "close", "volume", "trade_value"])
        for candle in candles:
            writer.writerow([
                candle["date"],
                candle["open"],
                candle["high"],
                candle["low"],
                candle["close"],
                candle["volume"],
                candle["trade_value"] or "",
            ])

    # 기간 정보 출력
    start_date = candles[0]["date"]
    end_date = candles[-1]["date"]
    print(f"  {result['exchange']}: {len(candles)}일치 ({start_date} ~ {end_date}) -> {filename}")


def main():
    print("=" * 60)
    print("TOKAMAK 일별 거래량 데이터 수집")
    print("=" * 60)

    exchanges = [
        ("upbit", get_upbit_candles),
        ("bithumb", get_bithumb_candles),
        ("biconomy", get_biconomy_candles),
        ("digifinex", get_digifinex_candles),
    ]

    print("\n데이터 수집 중...")
    for name, fetch_func in exchanges:
        try:
            result = fetch_func()
            filename = f"tokamak_{name}_daily.csv"
            save_to_csv(result, filename)
        except Exception as e:
            print(f"  {name}: 에러 - {e}")

    print("\n완료!")


if __name__ == "__main__":
    main()
```
