There are several reports in the literature that using Custom errors costs less gas than the required statement.
When comparing the cases of using revert and required, the gas cost was slightly lower when using revert. (This was not always the case, but generally was the case.)

We changed the code to use the revert statement when there was a gas cost reduction after comparing the gas cost when using Custom errors and when using the required statement.

In practice, the difference in gas cost between using revert and require is negligible. So we will not modify the code in either case, whether using Custom errors instead of require or using require instead of Custom errors.

The two cases above are compared in the _claim function of OperatorManagerV1_1.sol.

```solidity
function _claim(address token, address to, uint256 amount) internal {
        if(token == address(0)) {
            if(address(this).balance < amount) revert InsufficientBalanceError();
            (bool success, ) = to.call{value: amount}("");
            if (!success) revert TransferEthError();
        } else {
            // if (IERC20(token).balanceOf(address(this)) < amount) revert InsufficientBalanceError();
            require(IERC20(token).balanceOf(address(this)) >= amount, InsufficientBalanceError());

            IERC20(token).safeTransfer(to, amount);
        }
        emit Claimed(token, msg.sender, to, amount);
    }
```
